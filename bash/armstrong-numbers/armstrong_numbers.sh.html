1 #!/usr/bin/env bash
 2 
 3 main () {
 4     input="$1"
 5     if [[ ! "$input" =~ ^[0-9]+$ ]]; then
 6         echo "only integer inputs allowed"
 7         exit 1
 8     fi
 9 
10     length="${#input}"

12         pow=$(( ${input:i:1} ** length ))
13         ((result+=pow))
14     done
15 
16     if [[ $result == $input ]]; then
17         echo "true"
18     else
19         echo "false"
20     fi
21 }
22
